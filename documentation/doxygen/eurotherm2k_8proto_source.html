00054 ### \brief Controller configuration mode
00055 ##
00056 ## This mode enable setting a number of controller configuration settings 
00057 ## (see Communications Handbook PDF pages 5-25 onwards)
00058 ##
00059 ## We set the mode by sending a command 'IM2' and get out of the mode with the command 'IM0'
00060 ## While in configuration mode the controller makes itself address 0 so any configuration commands
00061 ## have to be sent to address = 0
00062 setConfMode {  InTerminator = ""; out "\x04\$1\$1\$2\$2\x02IM2\x03%6<xor>"; in "\x06"; @mismatch{ in "%(\$3:ERR.PROC)r"; }; }
00063 
00064 ## Escape from the configuration mode in to normal operation mode
00065 clearConfMode {  InTerminator = ""; out "\x040000\x02IM0\x03%6<xor>"; in "\x06"; @mismatch{ in "%(\$1:ERR.PROC)r"; }; }
